461  !> Interpolate tabulated rates from the table
462  !!
463  !! This function uses a lin-log interpolation to calculate
464  !! a given reaction rate (contained in \ref parameter_class::tabulated_rates_file)
465  !! An example entry of a tabulated rate looks like:
466  !!
467  !! \l_file{...
468  !! be7 n be6 tabln -1.06774e+01
469  !! 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 0.000e+00 1.323e-94 1.437e-76 1.129e-63 5.421e-54 1.868e-46 2.028e-40 2.890e-22 3.813e-13 1.201e-07 5.800e-04 2.562e-01 2.534e+01 1.669e+04 1.341e+06 3.232e+07 3.659e+08 2.502e+09 1.201e+10 }
470  !!
471  !! @note The function contains also a log-log interpolation that is commented out
472  !! and that can be replaced with the current lin-log interpolation.
473  !!
474  !! @warning This function should be changed in case of a different temperature grid
475  !! for tabulated rates
476  !!
477  !! @see tabulated_inter, temp_grid_tab, parameter_class::use_tabulated_rates,
478  !! parameter_class::tabulated_rates_file
479  !!
480  !! @author D. Martin
481  function tabulated_inter(rate,temp) result (tabr)
482  implicit none
483 
484  real(r_kind),dimension(:) :: rate !< Rate entries
485  real(r_kind) :: temp !< Temperature [GK]
486  real(r_kind) :: tabr !< Interpolated rate at temperature
487 
488  tabr = 0.0d0
489  if(tab_index(1) .eq. tab_index(2)) then
490  tabr = rate(tab_index(1))
491  else
492  if((rate(tab_index(1)).lt.1.d-100) .or. (rate(tab_index(2)).lt.1.d-100)) then
493  tabr = 0.0d0
494  else
495  !!! lin-log
496  tabr = rate(tab_index(1))*(rate(tab_index(2))/rate(tab_index(1)))**((temp-temp_grid_tab(tab_index(1)))&
497  /(temp_grid_tab(tab_index(2))-temp_grid_tab(tab_index(1))))
498  !!! log-log
499  !tabr = rate(tab_index(1))*(rate(tab_index(2))/rate(tab_index(1)))**(dlog10(temp/temp_grid_tab(tab_index(1)))/dlog10(temp_grid_tab(tab_index(2))/temp_grid_tab(tab_index(1))))
500  endif
501  endif
502 
503  return
