259 !! Blank lines as well as comment lines are skipped.

266 subroutine read_param(parfile)
267  use file_handling_class
268  implicit none
269 
270  character(*),intent(in) :: parfile
271  !
272  character(1000) :: line
273  character(param_name_len) :: param_name
274  character(2000) :: param_value
275  character(2), parameter :: blanks = " "//achar(9)
276  integer :: parfile_unit, istat, ieq, i1,i2,ln
277 
278  parfile_unit= open_infile(parfile)
279 
280  ln= 1
281  read_loop: do
282  read (parfile_unit,'(A)',iostat=istat) line
283  if (istat .ne. 0) exit ! end of file
284  i1= verify(line,blanks)
285  if(i1.eq.0) cycle ! skip blank lines
286  i2= verify(line,blanks,back=.true.)
287  line= line(i1:i2) ! trim tabs and spaces
288  if(line(1:1).eq.'#') cycle ! skip comments, which start with '#'
289  ieq= index(line,'=') ! look for an '=' sign
290  if(ieq.eq.0) then
291  call raise_exception("Could not read parameter file in line # "//&
292  trim(adjustl(int_to_str(ln)))//" :"//new_line("A")//&
293  trim(adjustl(line)) ,"read_param",340003)
294  endif
295  i2= verify(line(1:ieq-1),blanks,back=.true.)
296  if(i2.eq.0) then
297  call raise_exception("Could not read parameter file in line # "//&
298  trim(adjustl(int_to_str(ln)))//" :"//new_line("A")//&
299  trim(adjustl(line)) ,"read_param",340003)
300  endif
301  param_name= line(1:i2) ! parse param_name and param_value
302  i2= verify(line,blanks,back=.true.)
303  i1= ieq-1 + verify(line(ieq:i2),blanks//"=")
304  param_value= line(i1:i2)
305  !print '(A," = :",A,";")', trim(param_name), trim(param_value)
306  call set_param(param_name,param_value)
307  ln= ln + 1
308  enddo read_loop
309 
310  ! Check the parameter for consistency
311  call check_param
312 
313  ! close the file
314  close(parfile_unit)
315 
316  return

1248 !> Search for similar existing parameters.
1249 !!
1250 !! Takes a list of parameters that are separated by ":" and a comparison
1251 !! parameter and returns the most similar parameter from the list. Furthermore,
1252 !! a score is returned that gives a measure of the similarity of the strings.
1253 !! This function is only called if a user specified parameter was not included
1254 !! in the list of possible parameters.
1255 !!
1256 !! @author M. Reichert
1257 !! @date 02.11.22
1258 subroutine search_close_parameter(possible_pars,input_par,score,cl_par)
1259  implicit none
1260  character(*),intent(in) :: possible_pars !< All possible parameter, long string
1261  ! that is separated by ":"
1262  character(*),intent(in) :: input_par !< Parameter to compare
1263  character(*),intent(out) :: cl_par !< Most similar parameter

1268  real(r_kind) :: h_dist !< helping distance
1269  integer :: length !< length of closest word

1271 
1272  ms_par = "" ! Initialize empty
1273  cl_dist = 999 ! Initialize with large number
1274  length = 999 ! Initialize with large number
1275  helper_char = "" ! Initialize empty
1276  ! Start from 2 since first character is a ":"
1277  do k=2,len(possible_pars)
1278

1287  h_dist = levenshteindistance(trim(adjustl(helper_char)),trim(adjustl(input_par)))
1288  if (h_dist .lt. cl_dist) then
1289  cl_dist = h_dist
1290  ms_par = trim(adjustl(helper_char))
1291  end if
1292  ! Reset the parameter
1293  helper_char = ""
1294  end if
1295 
1296  end do
1297

1299  score = cl_dist
1300  cl_par = trim(adjustl(ms_par))

1305 !> Calculates the Levenshtein distance between two strings
1306 !!
1307 !! The Levenshtein distance gives a measure of the similarity between
1308 !! two strings. The similarity is expressed in terms of a float afterwards.
1309 !! This routine follows the pseudocode [here](https://en.wikipedia.org/wiki/Levenshtein_distance)
1310 !!
1311 !! @author M. Reichert
1312 !! @date 02.11.22
1313 function levenshteindistance(par1,par2) result(dist)
1314  implicit none
1315  character(*),intent(in) :: par1 !< Input string 1
1316  character(*),intent(in) :: par2 !< Input string 2
1317  real(r_kind) :: dist !< Measure of the similarity
1318  character(1) :: upper1 !< Helper character
1319  character(1) :: upper2 !< Helper character
1320  integer :: i, j, h !< Loop variables
1321  integer :: m, n !< Lengths of par1 and par2
1322  integer :: stat !< Allocation status
1323  real(r_kind) :: scost !< Substitution cost
1324  real(r_kind),dimension(:,:),allocatable :: d !< Helper matrix
1325 
1326  ! Get the length of the strings
1327  m = len(par1)
1328  n = len(par2)
1329 
1330  allocate(d(0:m,0:n),stat=stat)
1331  if (stat .ne. 0) call raise_exception('Allocation of "d" failed.',&
1332  "LevenshteinDistance",320001)
1333 
1334  ! Initialize with zeros
1335  d(:,:) = 0
1336 
1337  ! Fill the first column
1338  do i=1, m
1339  d(i, 0) = i
1340  end do
1341  ! Fill the first row
1342  do j=1, n
1343  d(0, j) = j
1344  end do
1345 
1346  do j=1,n
1347  do i=1,m
1348  if (par1(i:i) .eq. par2(j:j)) then
1349  scost = 0
1350  else
1351  ! Check if it is only a case insensitivity
1352  h = iachar(par1(i:i))
1353  if (h>= iachar("a") .and. h<=iachar("z") ) then
1354  upper1 = achar(iachar(par1(i:i))-32)
1355  else
1356  upper1 = par1(i:i)
1357  end if
1358  h = iachar(par2(j:j))
1359  if (h>= iachar("a") .and. h<=iachar("z") ) then
1360  upper2 = achar(iachar(par2(j:j))-32)
1361  else
1362  upper2 = par2(j:j)
1363  end if
1364 
1365  ! It is not as bad when its only a mismatch in the upper/lower case
1366  if (upper2 .eq. upper1) then
1367  scost = 0.2
1368  else
1369  scost = 1
1370  end if
1371  end if
1372  ! Calculate the total cost of substitutions and exchanges
1373  d(i,j) = min(d(i-1,j)+1, d(i, j-1) + 1, d(i-1, j-1) + scost)
1374  end do
1375  end do
1376 
1377  ! Return value
1378  dist = d(m,n)
1379 
1380 end function levenshteindistance
