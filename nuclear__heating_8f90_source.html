291 !>
292 !! Modifies the temperature and entropy to account for nuclear heating
293 !!
294 !! The temperature/entropy is solved (explicitely) in the same Newton-Raphson
295 !! scheme as used for the abundances in e.g.,
296 !! \ref timestep_module::advance_implicit_euler and
297 !! \ref timestep_module::advance_gear.
298 !!
299 !! There are 3 different heating modes:
300 !! <table>
301 !! <caption id="multi_row">Heating modes</caption>
302 !! <tr><th> Value <th> Evolved variable <th> Additional source term
303 !! <tr><td> 0 <td> No heating <td> -
304 !! <tr><td> 1 <td> Entropy <td> adiabatic
305 !! <tr><td> 2 <td> Temperature <td> trajectory
306 !! <tr><td> 3 <td> Temperature <td> adiabatic
307 !! </table>
308 !!
309 !! @author M. Reichert
310 !! @date 10.02.23
311 !! .
312 subroutine nuclear_heating_update(nr_count,rho,Ye,pf,Y_p,Y,entropy_p,entropy,T9_p,T9,T9_tr_p,T9_tr,timestep)
313  use global_class, only: net_size,isotope
314  use nucstuff_class, only: inter_partf
315  use parameter_class, only: engen, top_engen_every, unit, heating_frac, &
316  heating_mode, freeze_rate_temp, use_thermal_nu_loss
317  use thermal_neutrino_module, only: thermal_neutrinos
318  implicit none
319 
320  integer,intent(in) :: nr_count !< NR-iteration count
321  real(r_kind),intent(in) :: rho !< density
322  real(r_kind),intent(in) :: ye !< electron fraction
323  real(r_kind),dimension(0:net_size),intent(out) :: pf !< partition functions
324  real(r_kind),dimension(net_size),intent(in) :: y_p !< abundances
325  real(r_kind),dimension(net_size),intent(in) :: y !< abundances at the next step
326  real(r_kind),intent(in) :: entropy_p !< entropy
327  real(r_kind),intent(out) :: entropy !< entropy for the next step
328  real(r_kind),intent(in) :: t9_tr_p !< temperature of the trajectory/analytic (prev.)
329  real(r_kind),intent(in) :: t9_tr !< temperature of the trajectory/analytic
330  real(r_kind),intent(in) :: t9_p !< temperature at the previous step
331  real(r_kind),intent(inout) :: t9 !< temperature at the next step
332  real(r_kind),intent(in) :: timestep !< timestep (s)
333 
334  real(r_kind) :: mic2, mui, kbt !< auxiliary variables and constants
335  type(timmes_eos_state) :: state !< Timmes EOS state
336  real(r_kind) :: gi !< statistical weight
337  real(r_kind) :: twopi_h2c2 !< 2\pi/(hc)^2 [MeV^-2 cm^-2]
338  real(r_kind) :: entropy_inc !< entropy increment during single timestep due to nuclear heating
339  real(r_kind) :: engen_i !< generated energy (per baryon) due to abundance changes of species i
340  real(r_kind) :: t9_old !< last temperature difference
341  real(r_kind) :: new_tdiff !< Temperature difference
342  real(r_kind),save :: last_tdiff = 0!< last temperature difference
343  real(r_kind) :: abar !< average mass number
344 
345  real(r_kind) :: ysum,yasum, temp_next
346  integer :: i ! Loop variables
347  integer :: eos_status
348 
349  info_entry("nuclear_heating_update")
350 
351  ! Only heat for T9 > freeze_rate_temp
352  if (t9 .gt. freeze_rate_temp*1.000001)then
353 
354  ! Save the old temperature
355  t9_old = t9
356 
357  if (use_thermal_nu_loss) then
358  ysum= sum(y(1:net_size))
359  yasum= sum(y(1:net_size)*isotope(1:net_size)%mass)
360  abar= yasum / ysum
361  call thermal_neutrinos(abar, ye, t9, rho, timestep, qdot_th)
362  qdot_th = qdot_th * unit%hix ! erg/g -> MeV/baryon
363  else
364  qdot_th = 0d0
365  end if
366 
367  ! Different heating treatments
368  if (heating_mode .eq. 1) then
369  call nuclear_heating_entropy(rho,ye,pf,y_p,y,entropy_p,entropy,t9,eos_status)
370  elseif ((heating_mode .eq. 2) .or. (heating_mode .eq. 3)) then
371  call nuclear_heating_energy(rho,ye,pf,y_p,y,entropy_p,entropy,t9_p,t9,t9_tr_p,t9_tr,eos_status)
372  else
373  call raise_exception("Heating_mode not implemented, got "//int_to_str(heating_mode)//".",&
374  'nuclear_heating_update',310004)
375  end if
376

378  ! if the EOS fails

380  t9 = t9_p
381  else

386  if (new_tdiff .ne. 0) then

390  end if

399  ysum= sum(y(1:net_size))
400  yasum= sum(y(1:net_size)*isotope(1:net_size)%mass)
401  state%abar= yasum / ysum
402  call timmes_eos(ink,t9*1d9,rho,ye,state,eos_status)
403

405 
406  else

408  end if

415 !>
416 !! Modifies the temperature to account for nuclear heating
417 !!
418 !! Here, an additional equation is calculated for the temperature.
419 !! The temperature is calculated by:
420 !! \f[ \dot{T}_\mathrm{nuc} = \frac{1}{c_v} \sum_i M_i dY_i \f]
421 !! with the symbols defined as:
422 !! <pre style="line-height:.5">
423 !! - M_i : mass excess [MeV] (see \ref global_class::isotope\%mass_exc)
424 !! - cv : Specific heat capacity at constant volume
425 !! .
426 !! </pre>
427 !!
428 !! For heating_mode 2, the temperature is calculated by:
429 !! \f[ \dot{T}_\mathrm{nuc} = \dot{T}_\mathrm{nuc} + \dot{T}_\mathrm{tr} \f]
430 !! where \f$\dot{T}_\mathrm{tr}\f$ is the temperature change of the external conditions,
431 !! i.e., the trajectory or the analytic equation.
432 !! For heating_mode 3, the temperature is calculated by:
433 !! \f[ \dot{T}_\mathrm{nuc} = \dot{T}_\mathrm{nuc} + \dot{T}_\mathrm{ad} \f]
434 !! where \f$\dot{T}_\mathrm{ad}\f$ is the temperature change assuming adiabatic conditions.
435 !!
436 !! @author M. Reichert
437 !! @date 10.02.23
438 !!
439 !! @see [Lippuner & Roberts 2017](https://ui.adsabs.harvard.edu/abs/2017ApJS..233...18L/abstract),
440 !! [XNet](https://github.com/starkiller-astro/XNet)
441 !! .
442 subroutine nuclear_heating_energy(rho,Ye,pf,Y_p,Y,entropy_p,entropy,T9_p,T9,T9_tr_p,T9_tr,eos_status)
443  use global_class, only: net_size,isotope
444  use nucstuff_class, only: inter_partf
445  use parameter_class, only: engen, top_engen_every, unit, &
446  heating_frac, freeze_rate_temp, heating_mode
447  use expansion_module,only: expand_count
448  implicit none
449 
450  real(r_kind),intent(in) :: rho !< density
451  real(r_kind),intent(in) :: ye !< electron fraction
452  real(r_kind),dimension(0:net_size),intent(out) :: pf !< partition functions
453  real(r_kind),dimension(net_size),intent(in) :: y_p !< abundances
454  real(r_kind),dimension(net_size),intent(in) :: y !< abundances at the next step
455  real(r_kind),intent(in) :: entropy_p !< entropy
456  real(r_kind),intent(out) :: entropy !< entropy for the next step
457  real(r_kind),intent(inout) :: t9 !< temperature at the next step
458  real(r_kind),intent(in) :: t9_p !< temperature at the previous step
459  real(r_kind),intent(in) :: t9_tr_p !< temperature of the trajectory/analytic (prev.)
460  real(r_kind),intent(in) :: t9_tr !< temperature of the trajectory/analytic
461  integer,intent(out) :: eos_status !< status of the EOS call
462 
463  real(r_kind) :: mic2, mui, kbt !< auxiliary variables and constants
464  type(timmes_eos_state) :: state !< Timmes EOS state
465  real(r_kind) :: gi !< statistical weight
466  real(r_kind) :: twopi_h2c2 !< 2\pi/(hc)^2 [MeV^-2 cm^-2]
467  real(r_kind) :: entropy_inc !< entropy increment during single timestep due to nuclear heating
468  real(r_kind) :: tdot_i
469  real(r_kind) :: tdot
470  real(r_kind) :: add_incr
471 
472  real(r_kind) :: ysum,yasum, temp_next, cv
473  integer :: i ! Loop variables
474 
475 
476  ! Call the Timmes EOS to get the specific heat at constant volume
477  ysum= sum(y(1:net_size))
478  yasum= sum(y(1:net_size)*isotope(1:net_size)%mass)
479  state%abar= yasum / ysum
480  call timmes_eos(ink,t9*1d9,rho,ye,state,eos_status)
481 
482  ! If this fails, return
483  if (eos_status .ne. 0) then
484  return
485  end if
486 
487  ! Calculate the energy increment due to nuclear heating
488  tdot = 0
489  do i=1,net_size
490  tdot_i = -(isotope(i)%mass_exc)*(y(i)-y_p(i)) / (state%cv)
491  tdot = tdot + tdot_i
492  end do
493 
494  ! Radiate energy away
495  tdot = tdot - (qdot_bet+qdot_nu+qdot_th) / (state%cv)
496 
497 
498  ! Only heat when the temperature is above freeze_rate_temp
499  if (t9 .gt. freeze_rate_temp*1.000001)then
500 
501  if (heating_mode .eq. 2) then
502  ! Heating mode 2 follows the trajectory until it ends,
503  ! afterwards adiabatic expansion is assumed
504  if (expand_count .ge. 1) then
505  call timmes_eos(ins,entropy_p,rho,ye,state,eos_status)
506 
507  ! If this fails, return
508  if (eos_status .ne. 0) then
509  return
510  end if
511 
512  add_incr = state%t*1d-9 - t9_p
513  else
514  add_incr = (t9_tr-t9_tr_p)
515  end if
516  elseif (heating_mode .eq. 3) then
517  ! Heating mode 3 assumes adiabatic expansion
518  call timmes_eos(ins,entropy_p,rho,ye,state,eos_status)
519 
520  ! If this fails, return
521  if (eos_status .ne. 0) then
522  return
523  end if
524 
525  add_incr = state%t*1d-9 - t9_p
526  end if
527 
528  t9 = t9_p + (tdot*1e-9) + add_incr
529 
530  else
531  eos_status = 0
532  t9 = t9

540 !>
541 !! Modifies the temperature and ent_p to account for nuclear heating
542 !!
543 !! Here, an additional equation is calculated for the entropy.
544 !! The entropy is calculated by:
545 !! \f[ E_\mathrm{gen} = \sum_i (m_i c^2 + \mu_i + Z_i \mu_e) dY_i -\Delta q \f]
546 !! , where \f$ \mu_i \f$ is given by
547 !! \f[
548 !! \mu_i = -k_B T \log \left( \frac{g_i G_i}{\rho N_A Y_i}
549 !! \left(\frac{2 \pi m_ic^2 k_B T}{h^2 c^2}\right)^{1.5} \right)
550 !! \f]
551 !! and the nuclear mass defined by the atomic mass excess:
552 !! \f[ m_ic^2 = M_i +Am_uc^2 - Z_i m_e c^2 .\f]
553 !! The symbols are defined as:
554 !! <pre style="line-height:.5">
555 !! - M_i : mass excess [MeV] (see \ref global_class::isotope\%mass_exc)
556 !! - m_ic^2 : nuclear mass (see \ref global_class::isotope\%mass [MeV])
557 !! - kB : Boltzman constant (see \ref parameter_class::unit\%k_mev)
558 !! - g_i : 2 * spin + 1 ( see \ref global_class::isotope\%spin )
559 !! - G_i : normalized partition function of isotope i
560 !! - rho : Density [cgs]
561 !! - \f$mu_e\f$ : Electron chemical potential
562 !! - Z_i : Proton number of isotope i
563 !! - Navo : Avogadro constant (see \ref parameter_class::unit\%n_a)
564 !! - Y_i : Abundance [mol/g]
565 !! - h : Planck constant [MeV*s] (see \ref parameter_class::unit\%h_mevs)
566 !! - c : Speed of light [cm/s] (see \ref parameter_class::unit\%clight)
567 !! .
568 !! </pre>
569 !! The entropy is converted to a temperature via the equation of state at the
570 !! end of this subroutine. Furthermore, \f$ \Delta q \f$ is the energy loss due
571 !! neutrinos that won't thermalize (\ref parameter_class::heating_frac ,
572 !! \ref beta_decay_file , \ref neutrino_loss_file).
573 !!
574 !! @see
575 !! [Freiburghaus et al. 1999, Eq. 2](https://ui.adsabs.harvard.edu/abs/1999ApJ...525L.121F/abstract)
576 !! [Mueller 1986](https://ui.adsabs.harvard.edu/abs/1986A%26A...162..103M/abstract)
577 !!
578 !! @author O. Korobkin
579 !!
580 !! \b Edited:
581 !! - MR: 20.01.21 - Separated top contributor to independent subroutine
582 !! .
583 subroutine nuclear_heating_entropy(rho,Ye,pf,Y_p,Y,entropy_p,entropy,T9,eos_status)
584  use global_class, only: net_size,isotope
585  use nucstuff_class, only: inter_partf
586  use parameter_class, only: engen, top_engen_every, unit, &
587  heating_frac, freeze_rate_temp
588  implicit none
589 
590  real(r_kind),intent(in) :: rho !< density
591  real(r_kind),intent(in) :: ye !< electron fraction
592  real(r_kind),dimension(0:net_size),intent(out) :: pf !< partition functions
593  real(r_kind),dimension(net_size),intent(in) :: y_p !< abundances
594  real(r_kind),dimension(net_size),intent(in) :: y !< abundances at the next step
595  real(r_kind),intent(in) :: entropy_p !< entropy
596  real(r_kind),intent(out) :: entropy !< entropy for the next step
597  real(r_kind),intent(inout) :: t9 !< temperature at the next step
598  integer,intent(out) :: eos_status !< Status of the EOS call
599 
600  real(r_kind) :: mic2, mui, kbt !< auxiliary variables and constants
601  type(timmes_eos_state) :: state !< Timmes EOS state
602  real(r_kind) :: gi !< statistical weight
603  real(r_kind) :: twopi_h2c2 !< 2\pi/(hc)^2 [MeV^-2 cm^-2]
604  real(r_kind) :: entropy_inc !< entropy increment during single timestep due to nuclear heating
605  real(r_kind) :: engen_i !< generated energy (per baryon) due to abundance changes of species i
606  real(r_kind) :: etaele,etapos !< Electron and positron chemical potentials from eos
607  real(r_kind) :: mue !< Electron and positron chemical potentials
608 
609  real(r_kind) :: ysum,yasum, temp_next
610  integer :: i ! Loop variables
611 
612  info_entry("nuclear_heating_entropy")
613 
614  ! calculate partition functions for all nuclei
615  call inter_partf (t9, pf)
616 
617  temp_next= t9 * 1.d9
618 
619 
620  ! Get the electron chemical potential
621  call chempot(temp_next,rho,ye,etaele,etapos)
622  mue = etaele * unit%k_mev * temp_next
623  mue = mue + unit%mass_e
624  ! calculate <A> based on abundances of the next timestep; this is needed for the EOS later
625  ysum= sum(y(1:net_size))
626  yasum= sum(y(1:net_size)*isotope(1:net_size)%mass)
627  state%abar= yasum / ysum
628 
629  ! update the temperature, using the entropy increment
630  ! entropy = engen / T
631  ! [entropy] = [MeV/baryon] / [K] -> kB [MeV/K] / baryon
632  ! calculate energy generated per baryon:
633  !
634  ! engen = sum_i (m_ic^2 + \mu_i + Z_i \mu_e) dY_i - \Delta q
635  !
636  ! where the chemical potential is:
637  ! { g_i G_i 2 \pi m_ic^2 kB T 3/2 }
638  ! \mu_i = -kB T log { ------------- ( ---------------- ) }
639  ! { rho Navo Y_i h^2 c^2 }
640  !
641  ! M_i = isotope(i)%mass_exc [MeV]
642  ! m_ic^2 = isotope(i)%mass [MeV]
643  ! kB = 8.6173324d-11 [Mev/K] = unit%k_mev
644  ! kB T = 8.6173324d-11 [Mev/K] * T [K] -> [MeV]
645  ! mu_e = Electron chemical potential [MeV]
646  ! g_i = isotope(i)%spin * 2 + 1
647  ! G_i = normalized partition function of isotope i
648  ! rho = rho [cgs]
649  ! Navo = 6.0221367d23 [1/mol] = unit%n_a
650  ! Y_i = Y(i) [mol/g] ... me: but also [1/baryon]?
651  ! h = 4.135667516e-21 [MeV*s] = unit%h_mevs
652  ! c = 2.99792458d10 [cm/s] = unit%clight
653  !
654  ! [engen] = MeV/nuc * nuc/baryon = MeV / baryon
655 
656  engen = 0.d0
657  kbt = temp_next * unit%k_mev ! 8.617332d-11 * T [MeV]
658 
659  do i=1,net_size
660  if (y(i).gt.1d-25) then
661  mic2 = isotope(i)%mass*unit%mass_u - isotope(i)%p_nr * unit%mass_e + isotope(i)%mass_exc ! [MeV]
662  gi = 2d0*isotope(i)%spin + 1d0 ! statistical weight
663  twopi_h2c2 = 2d0*unit%pi/(unit%h_mevs*unit%clight)**2 ! 2\pi/(hc)^2 [MeV^-2 cm^-2]
664  mui = -kbt*log( gi*pf(i) / (rho*y(i)*unit%n_a) * &
665  (twopi_h2c2 * isotope(i)%mass*unit%mass_u * kbt)**1.5d0 )
666  ! note that Amuc2 is a approximation of mic2. It is used in order to be consistent with what is
667  ! used in NSE.
668 
669  engen_i = -(mic2 + mui + isotope(i)%p_nr*mue)*(y(i)-y_p(i)) ! energy generated by species i [MeV/baryon]
670  engen = engen + engen_i
671  endif
672  end do
673 
674  ! the total energy from nuc. reactions
675  engen_nuc = engen_nuc + engen / unit%hix ! [erg/g]
676 
677  if (t9 .gt. freeze_rate_temp*1.000001)then
678  ! use only a fraction (=heating_frac) of engen: the rest is radiated away, e.g., due to neutrino cooling
679  entropy_inc= (engen - (qdot_bet + qdot_nu + qdot_th)) / kbt ! divide by kB in order to get entropy in units kB per baryon
680  entropy= entropy_p + entropy_inc ! [kB/baryon]
681 
682  ! Prevent unphysical negative entropy
683  ! This mostly happens for regions where the EOS is not
684  ! valid in any case, i.e., very high densities.
685  if (entropy .lt. 0) then
686  ! Complain about it at high verbose levels
687  if (verbose_level .gt. 1) then
688  write(*,*) "Warning! Restricting entropy as it was negative: ", entropy
689  end if
690  entropy = 0
691  end if
692 
693  call timmes_eos(ins,entropy,rho,ye,state,eos_status)
694 
695  if (eos_status .ne. 0 ) then
696  continue
697  ! call raise_exception("The EOS did not converge, "//NEW_LINE("A")//&
698  ! "T ="//num_to_str(T9)//"GK, "//NEW_LINE("A")//&
699  ! "rho ="//num_to_str(rho)//"g/ccm,"//NEW_LINE("A")//&
700  ! "Ye ="//num_to_str(Ye), "nuclear_heating_update",&
701  ! 310003)
702  else
703  temp_next= state%t ! new temperature with updated entropy
704  t9 = temp_next / 1d9 ! new temperature in [GK]
705  end if
706  t9 = temp_next / 1d9 ! new temperature in [GK]
707  else
708  eos_status = 0
709  t9= t9
710  end if
711 
712  info_exit("nuclear_heating_entropy")
713 
714 
715 end subroutine nuclear_heating_entropy
